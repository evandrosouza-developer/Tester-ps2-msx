222void portXread(void)
223{
224 uint8_t mountstring[6]; //Used in usart_send_string()
225
226 //To be measured the real time from column *Y_ writing to reading, by putting an oscilloscope at pin A1
227 GPIO_BSRR(Xint_port) = Xint_pin_id; //Back to default condition ("1")
228
229 // Read the MSX keyboard X answer through GPIO pins X7:X0
230#if MCU == STM32F103
231 uint16_t wmsx_X;
232 wmsx_X = (gpio_port_read(X0_port) >> 6); //Read bits B15, B14, B13, B12, B9, B8, B7 and B6 (1111.0011.1100.0000 F3C0)
233 msx_X = (uint8_t)((wmsx_X >> 2) & 0xF0) | (uint8_t)(wmsx_X & 0x0F);
234#endif //#if MCU == STM32F103
235#if MCU == STM32F401
236 uint16_t tmp = (gpio_port_read(X0_port));
237 //Bits order on F4 are different!
238 //Calculate them are easer (and smaller) than find values in a 16 bits lookup table!
239 //Bit 7
240 msx_X = (tmp & (1 << MSX_X_BIT7)) ? 1 << 7 : 0;
241 //Bit 6
242 msx_X |= (tmp & (1 << MSX_X_BIT6)) ? 1 << 6 : 0;
243 //Bit 5
244 msx_X |= (tmp & (1 << MSX_X_BIT5)) ? 1 << 5 : 0;
245 //Bit 4
246 msx_X |= (tmp & (1 << MSX_X_BIT4)) ? 1 << 4 : 0;
247 //Bit 3
248 msx_X |= (tmp & (1 << MSX_X_BIT3)) ? 1 << 3 : 0;
249 //Bit 2
250 msx_X |= (tmp & (1 << MSX_X_BIT2)) ? 1 << 2 : 0;
251 //Bit 1
252 msx_X |= (tmp & (1 << MSX_X_BIT1)) ? 1 << 1 : 0;
253 //Bit 0
254 msx_X |= (tmp & (1 << MSX_X_BIT0)) ? 1 << 0 : 0;
255#endif //#if MCU == STM32F401
256
257 if (msx_X != msx_matrix[y_scan]) //Print info if ps2-MSX adapter has updated data of this y_scan
258 {
259 //Store the new value of msx_X to allow print the future keyboard changes
260 msx_matrix[y_scan] = msx_X;
261 //Read the result of this reading and mount it to a circular buffer string
262 //Print the changes through filling buffer that will be transfered via console in main
263 //Print y_scan and msx_X
264 string_append((uint8_t*)"Y", &pascal_string);
265 conv_uint8_to_2a_hex(y_scan, mountstring);
266 string_append(&mountstring[1], &pascal_string);
267 string_append((uint8_t*)" X", &pascal_string);
268 conv_uint8_to_2a_hex(msx_X, mountstring);
269 string_append((uint8_t*)mountstring, &pascal_string);
270 string_append((uint8_t*)"\r\n> ", &pascal_string);
271 }
272 if (!wait_flag)
273 {//Update here to next valid scan
274 y_scan++;
275 if (y_scan > end_scancount)
276 {
277 y_scan = init_scancount;
278 inactivity_cycles[scan_pointer] = init_inactivity_cycles[scan_pointer];
279 if(single_sweep)
280 {//Last colunm to scan: Back to paused state
281 single_sweep = false;
282 wait_flag = true; //Do this last one and, after portXread executed, stop single_sweep
283 } //if(single_sweep)
284 }
285 }
286 if (single_step)
287 {//Run this time and returns to paused state
288 single_step = false;
289 wait_flag = true;
290 }
291}
