93static void usb_cdc_set_state(usbd_device *usbd_dev, const uint16_t iface, const uint8_t ep)
94{
95 uint8_t buf[10];
96 struct usb_cdc_notification *notif = (void*)buf;
97 /* We echo signals back to host as notification */
98 notif->bmRequestType = 0xA1;
99 notif->bNotification = USB_CDC_NOTIFY_SERIAL_STATE;
100 notif->wValue = 0;
101 notif->wIndex = iface;
102 notif->wLength = 2;
103 buf[8] = 3U;
104 buf[9] = 0U;
105 usbd_ep_write_packet(usbd_dev, ep, buf, sizeof(buf));
106}

109static enum usbd_request_return_codes
110 cdcacm_control_request(usbd_device *usbd_dev, struct usb_setup_data *req, uint8_t **buf, uint16_t *len,
111 void (**complete)(usbd_device *usbd_dev, struct usb_setup_data *req))
112{
113 (void)complete;
114 (void)buf;
115 (void)usbd_dev;
116
117 switch (req->bmRequestType)
118 {
119 case SEND_ENCAPSULATED_COMMAND_bmRequestType: //0x21
120 if(req->bRequest == 0)
121 {
122 //send a RESPONSE_AVAILABLE notification
123 uint8_t loc_buf[4];
124 loc_buf[0] = 0x01;
125 loc_buf[1] = 0;
126 loc_buf[2] = 0;
127 loc_buf[3] = 0;
128 usbd_ep_write_packet(usbd_dev, EP_UART_COMM_IN, loc_buf, sizeof(uint32_t));
129 }
130 break; //case SEND_ENCAPSULATED_COMMAND_bmRequestType:
131 case GET_ENCAPSULATED_RESPONSE_bmRequestType: //0xA1
132 if(req->bRequest == GET_ENCAPSULATED_RESPONSE_bRequest) //0x01
133 {
134 usbd_ep_write_packet(usbd_dev, EP_UART_COMM_IN, &line_coding, sizeof(line_coding));
135 }
136 break; //case GET_ENCAPSULATED_RESPONSE_bmRequestType:
137 default:
138 break;
139 }
140 switch (req->bRequest)
141 {
142 case USB_CDC_REQ_SET_CONTROL_LINE_STATE: //0x22
143 /*
144 * This Linux cdc_acm driver requires this to be implemented
145 * even though it's optional in the CDC spec, and we don't
146 * advertise it in the ACM functional descriptor.
147 */
148
149 /* We echo signals back to host as notification. */
150 usb_cdc_set_state(usbd_dev, INTF_UART_COMM, EP_UART_COMM_IN);
151 return USBD_REQ_HANDLED;
152 break;
153 case USB_CDC_REQ_SET_LINE_CODING: //0x20
154 if (*len < sizeof(struct usb_cdc_line_coding))
155 return USBD_REQ_NOTSUPP;
156 switch(req->wIndex)
157 {
158 case 0:
159 return USBD_REQ_NOTSUPP; // Ignore on CON Port
160 break;
161 case 2:
162 usart_update_comm_param((struct usb_cdc_line_coding*)*buf);
163 return USBD_REQ_HANDLED;
164 break;
165 default:
166 return USBD_REQ_HANDLED;
167 break;
168 }
169 break;
170 case USB_CDC_REQ_GET_LINE_CODING:
171 *buf = (uint8_t *)&line_coding;
172 usbd_ep_write_packet(usbd_dev, EP_UART_COMM_IN, &line_coding, sizeof(line_coding));
173 return USBD_REQ_HANDLED;
174 break;
175 default:
176 return USBD_REQ_NOTSUPP;
177 break;
178 }
179 return USBD_REQ_NOTSUPP;
180}

183static void cdcacm_con_data_rx_cb(usbd_device *usbd_dev, uint8_t ep)
184{
185 (void)usbd_dev;
186 (void)ep;
187 char buf_con[USBD_DATA_BUFFER_SIZE];
188 uint16_t len_con, result = 0;
189
190 len_con = (uint16_t)usbd_ep_read_packet(usbd_dev, EP_CON_DATA_OUT, buf_con, USBD_DATA_BUFFER_SIZE);
191 if (len_con)
192 {
193 for(uint16_t i = 0; (i < len_con) && (result != 0xFFFF); i++)
194 while( (result = ring_put_ch(&con_rx_ring, buf_con[i])) == 0xFFFF) __asm("nop");
195 if(result > ((3 * (uart_tx_ring.bufSzMask + 1)) >> 2)) //X_OFF_TRIGGER
196 {
197 //Put EP in nak. On con_rx_ring read, check con_rx_ring room (X_ON_TRIGGER) to clear nak.
198 nak_cleared[EP_CON_DATA_OUT] = false;
199 usbd_ep_nak_set(usbd_dev, EP_CON_DATA_OUT, 1);
200 }
201 }
202}

205static void cdcacm_con_data_tx_cb(usbd_device *usbd_dev, uint8_t ep)
206{
207 (void)usbd_dev;
208 (void)ep;
209 char buf[USBD_DATA_BUFFER_SIZE];
210 uint16_t i, len, max_transf, qty_accepted, local_getptr;
211
212 len = QTTY_CHAR_IN(con_tx_ring);
213 if(len)
214 {
215 max_transf = (len > (USBD_DATA_BUFFER_SIZE - 1)) ? (USBD_DATA_BUFFER_SIZE - 1) : len;
216 local_getptr = con_tx_ring.get_ptr;
217
218 CHECK_XONXOFF_SENDNOW_START_WITH_OPEN_BRACKET
219 // Transmit char through usb console.
220 buf[0] = data;
221 for(i = 1; i < max_transf; i++)
222 {
223 buf[i] = con_tx_ring.data[local_getptr++];
224 local_getptr &= con_tx_ring.bufSzMask;
225 }
226 CHECK_XONXOFF_SENDNOW_CLOSE_BRACKET
227 else //else CHECK_XONXOFF_SENDNOW_START_WITH_OPEN_BRACKET
228 {
229 for(i = 0; i < max_transf; i++)
230 {
231 buf[i] = con_tx_ring.data[local_getptr++];
232 local_getptr &= con_tx_ring.bufSzMask;
233 }
234 } //else CHECK_XONXOFF_SENDNOW_START_WITH_OPEN_BRACKET
235
236 qty_accepted = usbd_ep_write_packet(usb_dev, EP_CON_DATA_IN, buf, max_transf);
237 //This following two passes warranties that con_tx_ring.get_ptr to be an atomic update.
238 con_tx_ring.get_ptr = (con_tx_ring.get_ptr + qty_accepted) & con_tx_ring.bufSzMask;
239 }
240 else //if(len)
241 { //As len = 0, set this EP to NAK requests, but first check XONXOFF SENDNOW
242 CHECK_XONXOFF_SENDNOW_START_WITH_OPEN_BRACKET
243 // Put char in console ring.
244 ring_put_ch(&con_tx_ring, data);
245 buf[0] = data;
246 usbd_ep_write_packet(usb_dev, EP_CON_DATA_IN, buf, 1);
247 CHECK_XONXOFF_SENDNOW_CLOSE_BRACKET //CHECK_XONXOFF_SENDNOW_START_WITH_OPEN_BRACKET
248 else //else CHECK_XONXOFF_SENDNOW_START_WITH_OPEN_BRACKET
249 usbd_ep_nak_set(usbd_dev, EP_CON_DATA_IN, 1);
250 }
251}

254static void cdcacm_uart_data_rx_cb(usbd_device *usbd_dev, uint8_t ep)
255{
256 (void)usbd_dev;
257 (void)ep;
258 uint8_t buf_uart[USBD_DATA_BUFFER_SIZE + 1];
259 uint16_t result;
260 uint32_t len;
261
262 len = (uint16_t)usbd_ep_read_packet(usbd_dev, EP_UART_DATA_OUT, (char*)buf_uart, USBD_DATA_BUFFER_SIZE);
263 if (len && ok_to_rx)
264 {
265 for(uint16_t i = 0; (i < len) && (result != 0xFFFF); i++)
266 while( (result = ring_put_ch(&uart_tx_ring, buf_uart[i])) == 0xFFFF)
267 do_dma_usart_tx_ring(len);
268 do_dma_usart_tx_ring(len);
269 if(result > ((3 * (uart_tx_ring.bufSzMask + 1)) >> 2))
270 {
271 //Put EP in nak. On uart_tx_ring read, check uart_rx_ring room (X_ON_TRIGGER) to clear nak.
272 set_nak_endpoint(EP_UART_DATA_OUT);
273 } //if(result > (3 * (uart_tx_ring.bufSzMask + 1) / 4))
274 } //if (len)
275}

302static void cdcacm_set_config(usbd_device *usbd_dev, uint16_t wValue)
303{
304 usb_configured = wValue;
305
306 //Console interface
307 usbd_ep_setup(usbd_dev, EP_CON_COMM_IN, USB_ENDPOINT_ATTR_INTERRUPT, COMM_PACKET_SIZE, NULL);
308 usbd_ep_setup(usbd_dev, EP_CON_DATA_OUT, USB_ENDPOINT_ATTR_BULK, USBD_DATA_BUFFER_SIZE, cdcacm_con_data_rx_cb);
309 usbd_ep_setup(usbd_dev, EP_CON_DATA_IN, USB_ENDPOINT_ATTR_BULK, USBD_DATA_BUFFER_SIZE, cdcacm_con_data_tx_cb);
310 //UART interface
311 usbd_ep_setup(usbd_dev, EP_UART_COMM_IN, USB_ENDPOINT_ATTR_INTERRUPT, COMM_PACKET_SIZE, NULL);
312 usbd_ep_setup(usbd_dev, EP_UART_DATA_OUT, USB_ENDPOINT_ATTR_BULK, USBD_DATA_BUFFER_SIZE, cdcacm_uart_data_rx_cb);
313 usbd_ep_setup(usbd_dev, EP_UART_DATA_IN, USB_ENDPOINT_ATTR_BULK, USBD_DATA_BUFFER_SIZE, cdcacm_uart_data_tx_cb);
314
315 usbd_register_control_callback(usbd_dev,
316 USB_REQ_TYPE_CLASS | USB_REQ_TYPE_INTERFACE,
317 USB_REQ_TYPE_TYPE | USB_REQ_TYPE_RECIPIENT,
318 cdcacm_control_request);
319
320 /* Notify the host that DCD is asserted.
321 * Allows the use of /dev/tty* devices on *BSD/MacOS
322 */
323 usb_cdc_set_state(usbd_dev, INTF_CON_COMM, EP_CON_DATA_OUT);
324 usb_cdc_set_state(usbd_dev, INTF_UART_COMM, EP_UART_DATA_OUT);
325}

328//It is used to start a communication pipe to allow the EP_??_DATA_IN callback continues send to host as needed.
329//It returns ring->get_ptr updated.
330void first_put_ring_content_onto_ep(struct sring *ring, uint8_t ep)
331{
332 if(usb_configured)
333 {
334 char buf[USBD_DATA_BUFFER_SIZE];
335 uint16_t i, len, max_transf, qty_accepted, local_getptr;
336
337 len = (ring->bufSzMask + 1 - ring->get_ptr + ring->put_ptr) & ring->bufSzMask;
338 local_getptr = ring->get_ptr; //update ring->get_ptr only at the end of the process
339 if(len)
340 {
341 clear_nak_endpoint(ep); //disable nak on ep
342 max_transf = (len > (USBD_DATA_BUFFER_SIZE - 1)) ? (USBD_DATA_BUFFER_SIZE - 1) : len;
343 for(i = 0; i < max_transf; i++)
344 {
345 buf[i] = ring->data[local_getptr++];
346 local_getptr &= ring->bufSzMask;
347 }
348 qty_accepted = usbd_ep_write_packet(usb_dev, ep, buf, max_transf);
349 //This following two passes warranties that ring->get_ptr to be an atomic update.
350 ring->get_ptr = (ring->get_ptr + qty_accepted) & ring->bufSzMask;
351 }
352 } //if(usb_configured)
353}

356void disable_usb(void)
357{
358 /* USB control register (USB_CNTR)
359Bit 1 PDWN: Power down
360This bit is used to completely switch off all USB-related analog parts if it is required to
361completely disable the USB peripheral for any reason. When this bit is set, the USB
362peripheral is disconnected from the transceivers and it cannot be used.
3630: Exit Power Down.
3641: Enter Power down mode.*/
365#if MCU == STM32F103
366 USB_CNTR_REG |= USB_CNTR_REG_PDWN;
367#endif //#if MCU == STM32F103
368# if (MCU == STM32F401)
369 // Explicitly disable DP pullup
370 OTG_FS_DCTL |= OTG_DCTL_SDIS;
371 rcc_periph_clock_disable(USB_RCC_CRC);
372# endif //# if (MCU == STM32F401)
373 rcc_periph_clock_disable(USB_RCC_OTGFS);
374}

406void cdcacm_init(void)
407{
408 usb_reset();
409
410#if MCU == STM32F401
411 // Enable peripherals
412 rcc_periph_clock_enable(USB_RCC_CRC);
413
414 gpio_mode_setup(OTG_FS_DM_PORT, GPIO_MODE_AF, GPIO_PUPD_NONE, OTG_FS_DM_PIN | OTG_FS_DP_PIN);
415 gpio_set_af(OTG_FS_DM_PORT, GPIO_AF10, OTG_FS_DM_PIN | OTG_FS_DP_PIN);
416#endif //#if MCU == STM32F401
417
418 usb_dev = usbd_init(&USB_DRIVER, &dev_desc, &config, usb_strings, NUM_STRINGS,
419 usbd_control_buffer, sizeof(usbd_control_buffer));
420
421#if MCU == STM32F401
422 //Disable VBUS sensing => https://github.com/libopencm3/libopencm3/pull/1256#
423 OTG_FS_GCCFG |= OTG_GCCFG_NOVBUSSENS | OTG_GCCFG_PWRDWN;
424 OTG_FS_GCCFG &= ~(OTG_GCCFG_VBUSBSEN | OTG_GCCFG_VBUSASEN);
425#if USART_PORT == USART1
426 //Now, if STM32F401 and USART1, recover A9 and A10 back to serial 1:
427 gpio_set_af(GPIO_BANK_USART_TX, GPIO_AF7, GPIO_PIN_USART_TX | GPIO_PIN_USART_RX);
428#endif //#if USART_PORT == USART1
429#endif //#if MCU == STM32F401
430
431 usbd_register_set_config_callback(usb_dev, cdcacm_set_config);
432
433 nvic_set_priority(USB_NVIC, IRQ_PRI_USB);
434 nvic_enable_irq(USB_NVIC);
435}
