57void serialno_read(uint8_t *s)
58{
59 // Fetch serial number from chip's unique ID
60 // Use the same serial number as the ST DFU Bootloader.
61# if (MCU == STM32F401)
62 uint16_t *uid_p = (uint16_t *)DESIG_UNIQUE_ID_BASE;
63#define OFFSET 3
64
65 uint64_t unique_id = ((uint64_t)(uid_p[1] + uid_p[5]) << 32) +
66 ((uint64_t)(uid_p[0] + uid_p[4]) << 16) +
67 uid_p[OFFSET];
68# endif
69
70# if (MCU == STM32F103)
71 uint32_t *uid_p = (uint32_t *)DESIG_UNIQUE_ID_BASE;
72 uint32_t unique_id = uid_p[0] + uid_p[1] + uid_p[2];
73# endif
74
75 //Convert each nibble to ASCII HEX
76 uint16_t i, ii;
77 for(i = 0; i < LEN_SERIAL_No; i++)
78 {
79 ii = LEN_SERIAL_No-1-i;
80 s[ii] = ((unique_id >> (4*i)) & 0xF) + '0';
81 if(s[ii] > '9')
82 s[ii] += 'A' - '9' - 1;
83 }
84 s[LEN_SERIAL_No] = 0;
85
86 return; //s
87}
