93void systick_setup(void)
94{
95 //Make sure systick doesn't interrupt more important interrupts
96 nvic_set_priority(NVIC_SYSTICK_IRQ, IRQ_PRI_SYSTICK);
97
98 systick_set_reload(SPEED_SYSTICK_RELOAD[INIT_SCAN_POINTER]-1);
99
100 systicks = 0; //systick_clear
101 ticks_keys = false;
102 wait_flag = false;
103 gpio_clear(CAPS_port, CAPS_pin_id);
104 gpio_set(KANA_port, KANA_pin_id);
105 caps_line = 0x0B; //Starts blinking
106 kana_line = 0x0B; //Starts blinking
107
108 systick_interrupt_enable();
109
110 /* Start counting. */
111 systick_counter_enable();
112
113 y_scan = init_scancount;
114}

134void sys_tick_handler(void)
135{
136 systicks++;
137
138 tickscaps++;
139 if(tickscaps >= SPEED_SYSTICK_DIVISOR[scan_pointer]) //~6000 on 32K
140 {
141 tickscaps = 0;
142
143 //C13 LED blink control
144 gpio_toggle(EMBEDDED_LED_port, EMBEDDED_LED_pin);
145
146 //CapsLock and Kana Line Control
147 switch (caps_line)
148 {
149 case 0x00:
150 gpio_clear(CAPS_port, CAPS_pin_id);
151 break;
152 case 0x01:
153 gpio_set(CAPS_port, CAPS_pin_id);
154 break;
155 case 0x0B:
156 gpio_toggle(CAPS_port, CAPS_pin_id);
157 break;
158 }
159 switch (kana_line)
160 {
161 case 0x00:
162 gpio_clear(KANA_port, KANA_pin_id);
163 break;
164 case 0x01:
165 gpio_set(KANA_port, KANA_pin_id);
166 break;
167 case 0x0B:
168 gpio_toggle(KANA_port, KANA_pin_id);
169 break;
170 }
171 } //if(tickscaps >= SPEED_SYSTICK_DIVISOR[scan_pointer])
172
173 if( !inactivity_cycles[scan_pointer] || (wait_flag || single_step || single_sweep) )
174 {
175 if (!wait_flag)
176 {
177 delay_qusec(TIM_HR, TIME_TO_READ_X_TABLE[delay_to_read_x_scan], portXread); //3.6us is the target. As the timer2 is ticking at 4MHz (250ns period)
178 //Put Y_Scan on port
179 GPIO_BSRR(Y_port) = y_bits[y_scan]; //Atomic GPIOA update => Update scan for the column
180 if (y_scan == init_scancount)
181 {
182 //clear Y_pin_id & Xint_pin_id
183 GPIO_BSRR(Y_port) = (Y_pin_id << 16); //To trig an oscilloscope to the scan start
184 }
185 //IMPORTANT: The update to next valid y_scan was moved to portXread (t_msxmap.cpp) to fix print mismatch
186 } //if (!wait_flag)
187 else
188 //To be capable of read even on wait
189 delay_qusec(TIM_HR, TIME_TO_READ_X_TABLE[delay_to_read_x_scan], portXread);
190 } // if( !inactivity_cycles[scan_pointer] || (wait_flag || single_step || single_sweep) )
191 else // if( !inactivity_cycles[scan_pointer] || (wait_flag || single_step || single_sweep) )
192 {
193 //It is here because it is not time to scan
194 //Update here to next valid scan
195 y_scan++;
196 if (y_scan > end_scancount)
197 {
198 y_scan = init_scancount;
199 inactivity_cycles[scan_pointer]--;
200 }
201 } //if( !inactivity_cycles[scan_pointer] || (wait_flag || single_step || single_sweep) )
202}
